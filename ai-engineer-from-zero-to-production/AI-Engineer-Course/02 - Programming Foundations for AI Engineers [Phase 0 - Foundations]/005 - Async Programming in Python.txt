# Async Programming in Python

## Overview
Almost every LLM API call is a network-bound I/O operation, which makes async programming the single most impactful performance technique for AI engineers. This lesson covers asyncio fundamentals, concurrent LLM calls with `asyncio.gather()`, async generators for streaming, and how async integrates with FastAPI and async HTTP clients like httpx.

## Key Points
- `async def` defines a coroutine — calling it returns a coroutine object, not a result; you must `await` it or run it with `asyncio.run()` to execute it
- `asyncio.gather(*coroutines)` runs multiple coroutines concurrently on a single thread — use this to call an LLM API N times in parallel instead of sequentially, reducing wall-clock time from N*latency to ~1*latency
- `asyncio.Semaphore(n)` limits concurrency — essential when hitting rate-limited APIs (e.g., OpenAI's requests-per-minute limits) to avoid 429 errors
- Async generators (`async def` with `yield`) are the correct abstraction for streaming LLM responses — FastAPI's `StreamingResponse` consumes them natively
- Never call synchronous blocking code (like `time.sleep()`, `requests.get()`, or file I/O) inside an async function — it blocks the entire event loop; use `asyncio.sleep()`, `httpx.AsyncClient`, and `aiofiles` instead
- `asyncio.create_task()` schedules a coroutine to run in the background without waiting for it — useful for fire-and-forget logging or telemetry calls

## Code Example
```python
from __future__ import annotations

import asyncio
import time
from typing import AsyncIterator

import httpx  # pip install httpx


# -------------------------------------------------------------------
# 1. Basic async/await — sequential vs concurrent
# -------------------------------------------------------------------
async def fake_llm_call(prompt: str, delay: float = 1.0) -> str:
    """Simulate a network-bound LLM API call."""
    await asyncio.sleep(delay)  # non-blocking sleep — event loop can run other tasks
    return f"Response to: '{prompt}'"


async def sequential_calls(prompts: list[str]) -> list[str]:
    results = []
    for prompt in prompts:
        result = await fake_llm_call(prompt)  # waits for each before starting next
        results.append(result)
    return results


async def concurrent_calls(prompts: list[str]) -> list[str]:
    """Run all calls at the same time — total time ≈ max(individual times)."""
    tasks = [fake_llm_call(prompt) for prompt in prompts]
    return list(await asyncio.gather(*tasks))


# -------------------------------------------------------------------
# 2. Rate-limited concurrent calls with Semaphore
# -------------------------------------------------------------------
async def rate_limited_calls(
    prompts: list[str],
    max_concurrent: int = 5,
) -> list[str]:
    semaphore = asyncio.Semaphore(max_concurrent)

    async def _call_with_limit(prompt: str) -> str:
        async with semaphore:          # at most max_concurrent calls in flight
            return await fake_llm_call(prompt, delay=0.5)

    return list(await asyncio.gather(*[_call_with_limit(p) for p in prompts]))


# -------------------------------------------------------------------
# 3. Async generator for streaming
# -------------------------------------------------------------------
async def stream_response(text: str, chunk_size: int = 3) -> AsyncIterator[str]:
    """Simulate a streaming LLM response — yields chunks as they arrive."""
    words = text.split()
    for i in range(0, len(words), chunk_size):
        chunk = " ".join(words[i : i + chunk_size])
        await asyncio.sleep(0.05)     # simulate network delay per chunk
        yield chunk + " "


async def consume_stream(stream: AsyncIterator[str]) -> str:
    parts: list[str] = []
    async for chunk in stream:
        print(f"  chunk: {chunk!r}", flush=True)
        parts.append(chunk)
    return "".join(parts)


# -------------------------------------------------------------------
# 4. Background task (fire-and-forget telemetry)
# -------------------------------------------------------------------
async def log_request(request_id: str) -> None:
    await asyncio.sleep(0.1)  # simulate async write to logging service
    print(f"  [telemetry] logged request {request_id}")


async def handle_request(prompt: str, request_id: str) -> str:
    asyncio.create_task(log_request(request_id))   # don't await — let it run in background
    return await fake_llm_call(prompt, delay=0.3)


# -------------------------------------------------------------------
# Demo
# -------------------------------------------------------------------
async def main() -> None:
    prompts = [f"Prompt {i}" for i in range(6)]

    print("Sequential:")
    t0 = time.perf_counter()
    await sequential_calls(prompts[:3])
    print(f"  Sequential time: {time.perf_counter() - t0:.2f}s")

    print("\nConcurrent:")
    t0 = time.perf_counter()
    results = await concurrent_calls(prompts[:3])
    print(f"  Concurrent time: {time.perf_counter() - t0:.2f}s")
    print(f"  Results: {results}")

    print("\nRate-limited (max 2 concurrent):")
    results_rl = await rate_limited_calls(prompts, max_concurrent=2)
    print(f"  Got {len(results_rl)} results")

    print("\nStreaming:")
    full_text = await consume_stream(
        stream_response("The quick brown fox jumps over the lazy dog near the river")
    )
    print(f"  Full: {full_text.strip()}")

    print("\nBackground task:")
    reply = await handle_request("Hello", request_id="req-abc123")
    print(f"  Reply: {reply}")
    await asyncio.sleep(0.2)   # let background task finish before exit


if __name__ == "__main__":
    asyncio.run(main())
```

## Notes
A very common mistake is mixing synchronous and asynchronous code without awareness. For example, using the `requests` library inside an async route handler will block the entire uvicorn event loop, effectively making your server single-threaded under load. Always use `httpx.AsyncClient` (a drop-in async replacement for `requests`) for outbound HTTP calls. If you must call a blocking function that you cannot replace, wrap it with `await asyncio.to_thread(blocking_fn, arg)` to run it in a thread pool without blocking the event loop. Finally, avoid creating bare event loops manually (`asyncio.get_event_loop().run_until_complete(...)`) — always use `asyncio.run()` at the top level, which properly handles loop creation and cleanup.
